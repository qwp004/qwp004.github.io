local function SafeHttpGet(url)
    local ok, result = pcall(function()
        return HttpService:GetAsync(url)
    end)
    if ok and result and result ~= "" then
        return result
    end
    
    local ok2, result2 = pcall(function()
        return game:HttpGet(url)
    end)
    if ok2 and result2 and result2 ~= "" then
        return result2
    end
    
    local ok3, result3 = pcall(function()
        if request then
            local resp = request({Url = url, Method = "GET"})
            if resp and resp.Body then
                return resp.Body
            end
        end
        return nil
    end)
    if ok3 and result3 and result3 ~= "" then
        return result3
    end
    
    return nil
end

local function Base64Decode(str)
    local ok, result = pcall(function()
        return HttpService:DecodeBase64(str)
    end)
    if ok and result then
        return result
    end
    
    local b = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/'
    str = string.gsub(str, '[^'..b..'=]', '')
    local result2 = {}
    
    local function charVal(idx)
        local ch = string.sub(str, idx, idx)
        if ch == '=' or ch == '' then return 0 end
        local pos = string.find(b, ch, 1, true)
        if pos then return pos - 1 end
        return 0
    end
    
    for i = 1, #str, 4 do
        local c1 = charVal(i)
        local c2 = charVal(i+1)
        local c3 = charVal(i+2)
        local c4 = charVal(i+3)
        
        local b1 = bit32.band(bit32.bor(bit32.lshift(c1, 2), bit32.rshift(c2, 4)), 0xFF)
        local b2 = bit32.band(bit32.bor(bit32.lshift(bit32.band(c2, 0x0F), 4), bit32.rshift(c3, 2)), 0xFF)
        local b3 = bit32.band(bit32.bor(bit32.lshift(bit32.band(c3, 0x03), 6), c4), 0xFF)
        
        table.insert(result2, string.char(b1))
        if string.sub(str, i+2, i+2) ~= '=' then
            table.insert(result2, string.char(b2))
        end
        if string.sub(str, i+3, i+3) ~= '=' then
            table.insert(result2, string.char(b3))
        end
    end
    local decoded = table.concat(result2)
    if decoded and #decoded > 0 then
        return decoded
    end
    
    return nil
end

local function XORDecryptBytes(encryptedStr, key)
    local keyLen = #key
    local len = #encryptedStr
    local batchSize = 4096
    local result = {}
    
    for i = 1, len, batchSize do
        local endIdx = math.min(i + batchSize - 1, len)
        local chunk = {}
        for j = i, endIdx do
            local byte = string.byte(encryptedStr, j)
            local keyByte = string.byte(key, ((j - 1) % keyLen) + 1)
            chunk[#chunk + 1] = string.char(bit32.bxor(byte, keyByte))
        end
        result[#result + 1] = table.concat(chunk)
    end
    
    return table.concat(result)
end

local function VerifyKey(inputKey)
    if not inputKey or inputKey == "" then
        return false, "请输入卡密"
    end

    local response = SafeHttpGet(CONFIG.KeyGistUrl)
    if not response then
        return false, "无法连接验证服务器，请检查网络"
    end

    response = response:match("^%s*(.-)%s*$")

    local directJson = nil
    pcall(function()
        local testOk, testData = pcall(function()
            return HttpService:JSONDecode(response)
        end)
        if testOk and testData then
            directJson = testData
        end
    end)

    local keyDataJson = nil
    if not directJson then
        local encStr = response:match('KEY_DATA%s*=%s*"([^"]+)"')
        if encStr then
            local decoded = Base64Decode(encStr)
            if decoded then
                local jOk, jData = pcall(function()
                    return HttpService:JSONDecode(decoded)
                end)
                if jOk and jData then
                    keyDataJson = jData
                end
            end
        end
    end

    local data = directJson or keyDataJson
    if not data then
        return false, "卡密数据解析失败"
    end

    local keyData = nil

    if data.keys and data.keys[inputKey] then
        keyData = data.keys[inputKey]
    end

    if not keyData and type(data) == "table" then
        for _, item in ipairs(data) do
            if item.code == inputKey then
                keyData = item
                break
            end
        end
    end

    if not keyData then
        return false, "卡密无效"
    end

    local used = keyData.used or false
    local machine = keyData.machine or ""
    if used == true and machine ~= "" then
        local machineId = GetMachineId()
        if machine ~= machineId then
            return false, "卡密已被其他设备使用"
        end
    end

    local duration = keyData.duration
    if type(duration) == "number" and duration > 0 then
        if keyData.created then
            local createdTime = tonumber(keyData.created)
            if createdTime then
                local expireTime = createdTime + (duration * 86400)
                if os.time() > expireTime then
                    return false, "卡密已过期"
                end
            end
        end
    elseif keyData.expire then
        local expireTime = tonumber(keyData.expire)
        if expireTime and os.time() > expireTime then
            return false, "卡密已过期"
        end
    end

    return true, "验证成功"
end

        end

end

                saved = val

    end)

        return saved

    pcall(function()
        if readfile then
            local val = readfile(CONFIG.SaveKeyFlag .. ".txt")
            if type(val) == "string" and val ~= "" then
                saved = val
            end
        end
    end)
    return saved
end

local function MakeDraggable(frame, handle)
    local dragging = false
    local dragInput = nil
    local dragStart = Vector3.new(0, 0, 0)
    local startPos = UDim2.new()

    handle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1
            or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = frame.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    handle.InputChanged:Connect(function(input)
        if dragging
            and (input.UserInputType == Enum.UserInputType.MouseMovement
                or input.UserInputType == Enum.UserInputType.Touch) then
            local delta = input.Position - dragStart
            frame.Position = UDim2.new(
                startPos.X.Scale, startPos.X.Offset + delta.X,
                startPos.Y.Scale, startPos.Y.Offset + delta.Y
            )
        end
    end)
end

local function LoadEncryptedScript(statusCallback)
    local function updateStatus(msg)
        print("[加载器] " .. msg)
        if statusCallback then
            pcall(statusCallback, msg)
        end
    end
    
    updateStatus("正在下载脚本数据...")
    local encryptedData = SafeHttpGet(CONFIG.ScriptGistUrl)
    if not encryptedData then
        updateStatus("❌ 无法加载脚本数据")
        return false, "无法加载脚本数据"
    end
    updateStatus("已下载 " .. #encryptedData .. " 字节，正在解析...")
    
    local b64str = nil
    local patterns = {
        'KEY%s*=%s*"([^"]+)"',
        'SCRIPT_DATA%s*=%s*"([^"]+)"',
        'data%s*=%s*"([^"]+)"',
        'encrypted%s*=%s*"([^"]+)"',
    }
    for _, pat in ipairs(patterns) do
        local match = encryptedData:match(pat)
        if match and #match > 100 then
            b64str = match
            break
        end
    end
    
    if not b64str then
        local cleanData = encryptedData:match("^%s*(.-)%s*$")
        if #cleanData > 100 and not cleanData:find("\n") then
            b64str = cleanData
        end
    end
    
    if not b64str then
        updateStatus("❌ 脚本数据格式错误")
        return false, "脚本数据格式错误"
    end
    updateStatus("解析Base64数据 (" .. #b64str .. " 字符)...")
    
    local decoded = Base64Decode(b64str)
    if not decoded then
        updateStatus("❌ Base64解码失败")
        return false, "脚本解码失败"
    end
    updateStatus("Base64解码完成，正在解密...")
    
    -- 让UI有机会更新
    task.wait(0.1)
    
    local decrypted = XORDecryptBytes(decoded, CONFIG.EncryptKey)
    if not decrypted or decrypted == "" then
        updateStatus("❌ 解密失败")
        return false, "脚本解密失败"
    end
    updateStatus("解密完成 (" .. #decrypted .. " 字符)，正在编译...")
    
    -- 让UI有机会更新
    task.wait(0.1)
    
    local fn, err = loadstring(decrypted)
    if not fn then
        updateStatus("❌ 编译失败: " .. tostring(err))
        return false, "脚本编译失败: " .. tostring(err)
    end
    updateStatus("编译成功，正在执行...")
    
    local execOk, execErr = pcall(fn)
    if not execOk then
        updateStatus("❌ 执行失败: " .. tostring(execErr))
        return false, "脚本执行失败: " .. tostring(execErr)
    end
    updateStatus("✅ 脚本执行成功！")
    
    return true
end
